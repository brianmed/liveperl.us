use Mojolicious::Lite;

get '/' => "index";

# Authenticate based on name parameter
under sub {
  my $self = shift;

  # Authenticated
  my $name = $self->param('name') || '';
  return 1 if $name eq 'Bender';

  # Not authenticated
  $self->render('denied');
  return undef;
};

# Only reached when authenticated
get '/auth' => 'auth';

app->start;
