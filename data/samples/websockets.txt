use Mojolicious::Lite;

websocket '/echo' => sub {
  my $self = shift;
  $self->on(json => sub {
    my ($self, $hash) = @_;
    $hash->{msg} = "echo: $hash->{msg}";
    $self->send({json => $hash});
  });
};

get '/' => 'index';

app->start;
