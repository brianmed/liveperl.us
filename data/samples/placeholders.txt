#!/usr/local/bin/perl

use Mojolicious::Lite;
 
get '/' => sub {
    my $self = shift;
 
    my $name  = $self->param("name");
    $self->render(template => "slash", name => $name, extra => "");
};
 
get '/:name' => sub {
    my $self = shift;
 
    my $name  = $self->param("name");
    $self->render(template => "slash", name => $name, extra => " ... via placeholder");
};
 
app->start;
